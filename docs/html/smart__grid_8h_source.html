251 current_time += time_step / 3600.0;
252 if (current_time >= 24.0) current_time -= 24.0; // Cycle de 24h
253
254 // Mettre à jour la production et la demande
255 double time_factor = 0.5 * (1.0 + sin(M_PI * current_time / 12.0)); // Variation journalière
256 for (auto& producer : producers) {
257 producer.update_output(current_time);

259 for (auto& consumer : consumers) {
260 consumer.update_demand(current_time);
261 }
262
263 // Calculer l'équilibre offre/demande
264 double total_production = 0.0;
265 for (const auto& producer : producers) {
266 total_production += producer.current_output;
267 }
268
269 double total_demand = 0.0;
270 for (const auto& consumer : consumers) {
271 total_demand += consumer.demand;
272 }
273
274 double imbalance = total_production - total_demand;
275 double energy = 0.0;
276 if (imbalance > 0) {
277 // Excédent : stocker dans la batterie
278 battery.charge(imbalance * (time_step / 3600.0), time_step);
279 energy = imbalance;
280 purchase_energy = 0.0;
281 } else {
282 // Déficit : utiliser la batterie
283 double energy_needed = -imbalance * (time_step / 3600.0);
284 double energy_from_battery = battery.discharge(energy_needed, time_step);
285 if (energy_from_battery < energy_needed) {
286 energy = energy_needed - energy_from_battery;
287 purchase_energy = energy;
288 // Acheter de l'énergie au réseau principal
289 std::cout << "Purchasing additional energy from the grid : "
290 << energy << " kWh" << std::endl;
291 }
292 }
